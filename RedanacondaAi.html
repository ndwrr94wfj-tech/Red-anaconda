            const key = veniceKey.value;

        function addMessage(author, text, type) {

            div.className = `message ${type}`;

            

            const time = new Date().toLocaleTimeString();

            

            // Format code blocks

            text = text.replace(/```(\w+)?\n([\s\S]*?)```/g, '<div class="code-block"><code>$2</code></div>');
