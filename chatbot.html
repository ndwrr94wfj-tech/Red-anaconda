        function sendQuick(text) {

            userInput.value = text;

            sendMessage();

        }

        

        async function sendMessage() {

            const text = userInput.value.trim();

            if (!text) return;

            

            // Add user message

            addMessage(text, 'user');

            userInput.value = '';

            

            // Show typing

            typingIndicator.classList.add('active');

            sendBtn.disabled = true;

            

            const model = document.getElementById('modelSelect').value;

            const apiKey = document.getElementById('apiKey').value;

            

            let response;

            

            if (model === 'demo' || !apiKey) {

                // Demo mode - use knowledge base

                await simulateTyping();

                response = getDemoResponse(text);

            } else {

                // Real API mode

                try {

                    response = await callOpenRouter(text, model, apiKey);

                } catch (err) {

                    response = `**Erreur API**: ${err.message}<br><br>Retour en mode démo.`;

                }

            }

            

            // Hide typing and add response

            typingIndicator.classList.remove('active');

            addMessage(response, 'bot');

            sendBtn.disabled = false;

            

            // Scroll to bottom

            chatMessages.scrollTop = chatMessages.scrollHeight;

        }

        

        function addMessage(text, sender) {

            const div = document.createElement('div');

            div.className = `message ${sender}`;

            div.innerHTML = `

                <div class="message-avatar">${sender === 'bot' ? '🤖' : '👤'}</div>

                <div class="message-bubble">${formatMessage(text)}</div>

            `;

            chatMessages.appendChild(div);

            chatMessages.scrollTop = chatMessages.scrollHeight;

        }

        

        function formatMessage(text) {

            // Format code blocks

            text = text.replace(/```(\w+)?\n([\s\S]*?)```/g, '<div class="code-block"><code>$2</code></div>');

            // Format inline code

            text = text.replace(/`([^`]+)`/g, '<code style="background: rgba(255,255,255,0.1); padding: 2px 5px; border-radius: 3px;">$1</code>');

            // Format bold

            text = text.replace(/\*\*(.*?)\*\*/g, '<strong style="color: #e94560;">$1</strong>');

            // Format newlines

            text = text.replace(/\n/g, '<br>');

            return text;

        }

        

        async function simulateTyping() {

            await sleep(1000 + Math.random() * 1000);

        }

        

        function getDemoResponse(text) {

            const lower = text.toLowerCase();

            

            if (lower.includes('osint') || lower.includes('c\'est quoi')) {

                return osintKnowledge['osint'];

            } else if (lower.includes('ip') || lower.includes('adresse')) {

                return osintKnowledge['ip'];

            } else if (lower.includes('dork') || lower.includes('google')) {

                return osintKnowledge['dorks'];

            } else if (lower.includes('username') || lower.includes('pseudo')) {

                return osintKnowledge['username'];

            } else if (lower.includes('email') || lower.includes('mail')) {

                return osintKnowledge['email'];

            } else {

                return osintKnowledge['default'];

            }

        }

        

        async function callOpenRouter(message, model, apiKey) {

            const response = await fetch('https://openrouter.ai/api/v1/chat/completions', {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json',

                    'Authorization': `Bearer ${apiKey}`,

                    'HTTP-Referer': window.location.href,

                    'X-Title': 'Red Anaconda OSINT'

                },

                body: JSON.stringify({

                    model: model,

                    messages: [

                        {

                            role: 'system',

                            content: 'Tu es un assistant OSINT expert. Tu aides les utilisateurs avec des techniques de renseignement en sources ouvertes de manière éthique et légale. Tu ne aides jamais à accéder illégalement à des données privées.'

                        },

                        {

                            role: 'user',

                            content: message

                        }

                    ]

                })

            });

            

            const data = await response.json();

            return data.choices[0].message.content;

        }
