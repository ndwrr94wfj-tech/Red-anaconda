        function checkUsername() {

            const user = document.getElementById('username').value.trim();

            const results = document.getElementById('userResults');

                {name: 'TikTok', url: `https://tiktok.com/@${user}`},

                {name: 'Reddit', url: `https://reddit.com/user/${user}`},

                {name: 'YouTube', url: `https://youtube.com/@${user}`},

            ];

                </div>`;

            });

            results.innerHTML = html;

        }

        // Google Dorks

        function generateDorks() {

            const target = document.getElementById('dorkTarget').value.trim();

            const type = document.getElementById('dorkType').value;

            const results = document.getElementById('dorkResults');

            if (!target) return;

            

            results.style.display = 'block';

            let dorks = [];

            

            if (type === 'person') {

                dorks = [

                    `"${target}" filetype:pdf`,

                    `"${target}" site:linkedin.com`,

                    `"${target}" site:facebook.com`,

                    `"${target}" intitle:"resume" OR intitle:"cv"`,

                    `intext:"${target}" filetype:doc`,

                ];

            } else if (type === 'domain') {

                dorks = [

                    `site:${target} filetype:pdf`,

                    `site:${target} inurl:admin`,

                    `site:${target} intitle:"index of"`,

                    `site:${target} ext:conf OR ext:config`,

                    `site:pastebin.com "${target}"`,

                ];

            } else {

                dorks = [

                    `"${target}"`,

                    `"${target}" site:pastebin.com`,

                    `"${target}" filetype:txt`,

                    `intext:"${target}"`,

                ];

            }

            

            let html = '<div class="result-label">Generated Dorks:</div>';

            dorks.forEach(dork => {

                const link = `https://www.google.com/search?q=${encodeURIComponent(dork)}`;

                html += `<div class="result-item" style="font-family: monospace; font-size: 0.9em;">

                    <a href="${link}" target="_blank" style="color: #e94560; text-decoration: none;">${dork}</a>

                </div>`;

            });

            results.innerHTML = html;

        }


        // Password Generator

        function generatePassword() {

            const length = parseInt(document.getElementById('passLength').value);

            const results = document.getElementById('passResults');

            const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*';

            let password = '';

            for (let i = 0; i < length; i++) {

                password += chars.charAt(Math.floor(Math.random() * chars.length));

            }

            

            results.style.display = 'block';

            results.innerHTML = `

                <div class="result-item">

                    <div class="result-label">Generated Password</div>

                    <div class="result-value" style="font-family: monospace; font-size: 1.2em; letter-spacing: 2px;">${password}</div>

                    <button onclick="navigator.clipboard.writeText('${password}')" style="margin-top: 10px;">Copy</button>

                </div>

            `;

        }
